     ```typescript
     export function deriveFlowBadges(workflowStatus: string): FlowBadge[] {
       // Cumulative: later stages include earlier flow badges
       const flows: FlowBadge[] = [];
       const s = workflowStatus;
       // Portal zugesendet: once portal access has been sent or any later stage
       if (s !== 'created') {
         flows.push('Portal zugesendet');
       }
       // 1. Anschreiben: once creditor contact is active or later
       if (s === 'creditor_contact_active' || s === 'completed') {
         flows.push('1. Anschreiben');
       }
       // 2. Anschreiben and Insolvenzantrag are future stages; only show on 'completed'
       if (s === 'completed') {
         flows.push('2. Anschreiben');
         flows.push('Insolvenzantrag');
       }
       return flows;
     }
     ```
