Create `MandantenPortalDesign/src/store/api/reviewApi.ts` following the injectEndpoints pattern from clientsApi.ts:
```typescript
import { baseApi } from './baseApi';
import type { ReviewQueueResponse } from '../../app/types';

interface ReviewQueueParams {
  page?: number;
  limit?: number;
}

export const reviewApi = baseApi.injectEndpoints({
  endpoints: (builder) => ({
    getReviewQueue: builder.query<ReviewQueueResponse, ReviewQueueParams>({
      query: ({ page = 1, limit = 25 } = {}) => ({
        url: '/api/agent-review/available-clients',
        params: { page, limit },
      }),
      providesTags: ['ReviewQueue'],
    }),
  }),
});

export const { useGetReviewQueueQuery } = reviewApi;
```
