import { baseApi } from './baseApi';
import type {
  ReviewDataResponse,
  SaveCorrectionPayload,
  SaveCorrectionResponse,
  CompleteReviewResponse,
} from '../../app/types';

export const reviewApi = baseApi.injectEndpoints({
  endpoints: (builder) => ({
    getReviewQueue: builder.query<any, { page?: number; limit?: number; priority?: string }>({
      query: ({ page = 1, limit = 25, priority } = {}) => ({
        url: '/api/agent-review/available-clients',
        params: { page, limit, ...(priority && priority !== 'all' ? { priority } : {}) },
      }),
      providesTags: ['ReviewQueue'],
    }),
    getReviewData: builder.query<ReviewDataResponse, string>({
      query: (clientId) => `/api/agent-review/${clientId}`,
      providesTags: (_result, _error, clientId) => [{ type: 'ReviewData', id: clientId }],
    }),
    saveReviewCorrection: builder.mutation<SaveCorrectionResponse, { clientId: string; body: SaveCorrectionPayload }>({
      query: ({ clientId, body }) => ({
        url: `/api/agent-review/${clientId}/correct`,
        method: 'POST',
        body,
      }),
      invalidatesTags: (_result, _error, { clientId }) => [{ type: 'ReviewData', id: clientId }],
    }),
    completeReview: builder.mutation<CompleteReviewResponse, string>({
      query: (clientId) => ({
        url: `/api/agent-review/${clientId}/complete`,
        method: 'POST',
      }),
      invalidatesTags: ['ReviewQueue'],
    }),
  }),
});

export const {
  useGetReviewQueueQuery,
  useGetReviewDataQuery,
  useSaveReviewCorrectionMutation,
  useCompleteReviewMutation,
} = reviewApi;

**reviewUiSlice.ts (NEW):**
Create `MandantenPortalDesign/src/store/slices/reviewUiSlice.ts`. This slice tracks client-side review UI state:

```typescript
import { createSlice, PayloadAction } from '@reduxjs/toolkit';
import type { ReviewAction } from '../../app/types';

interface CreditorReviewAction {
  creditorId: string;
  action: ReviewAction;
  skipReason?: string;
  corrections?: Record<string, unknown>;
}

interface ReviewUiState {
  currentCreditorIndex: number;
  actions: Record<string, CreditorReviewAction>; // keyed by creditorId
  skipReasonsEnabled: boolean;
}

const initialState: ReviewUiState = {
  currentCreditorIndex: 0,
  actions: {},
  skipReasonsEnabled: true,
};

export const reviewUiSlice = createSlice({
  name: 'reviewUi',
  initialState,
  reducers: {
    setCurrentCreditorIndex(state, action: PayloadAction<number>) {
      state.currentCreditorIndex = action.payload;
    },
    recordReviewAction(state, action: PayloadAction<CreditorReviewAction>) {
      state.actions[action.payload.creditorId] = action.payload;
    },
    setSkipReasonsEnabled(state, action: PayloadAction<boolean>) {
      state.skipReasonsEnabled = action.payload;
    },
    resetReviewState() {
      return initialState;
    },
  },
});

export const {
  setCurrentCreditorIndex,
  recordReviewAction,
  setSkipReasonsEnabled,
  resetReviewState,
} = reviewUiSlice.actions;

export const selectCurrentCreditorIndex = (state: { reviewUi: ReviewUiState }) => state.reviewUi.currentCreditorIndex;
export const selectReviewActions = (state: { reviewUi: ReviewUiState }) => state.reviewUi.actions;
export const selectSkipReasonsEnabled = (state: { reviewUi: ReviewUiState }) => state.reviewUi.skipReasonsEnabled;
```
