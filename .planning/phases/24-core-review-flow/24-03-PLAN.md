**Dialog content layout:**

6. **Handle completion with auto-navigation to next case:**
   Per CONTEXT.md: "After completing a review, auto-load the next case from the queue rather than redirecting to the queue page."

   ```typescript
   const navigate = useNavigate();
   const { data: queueData } = useGetReviewQueueQuery({ page: 1, limit: 1 }, { skip: !completionPending });

   const handleComplete = () => {
     // After successful completion, check for next available case
     // Use a small delay to allow queue to update after completion
     setTimeout(() => {
       if (queueData?.clients?.length > 0) {
         const nextClient = queueData.clients[0];
         if (nextClient.id !== clientId) {
           // Navigate to next case
           dispatch(resetReviewState());
           navigate(`/review/${nextClient.id}`, { replace: true });
           toast.success('Nachster Fall wird geladen...');
         } else {
           // Same client returned (shouldn't happen after completion) — go to queue
           dispatch(resetReviewState());
           navigate('/review', { replace: true });
         }
       } else {
         // No more cases — go to queue
         dispatch(resetReviewState());
         navigate('/review', { replace: true });
         toast.info('Keine weiteren Falle in der Queue');
       }
     }, 500);
   };
   ```

   Better approach: Trigger a fresh queue fetch after completion using `refetch`:
   ```typescript
   const [completionDone, setCompletionDone] = useState(false);

   const handleComplete = async () => {
     setCompletionDone(true);
     // ReviewSummaryDialog already called POST /complete successfully
     // Now fetch fresh queue to find next case
     try {
       const result = await fetch('/api/agent-review/available-clients?page=1&limit=1', {
         headers: { 'Authorization': `Bearer ${localStorage.getItem('admin_token')}` },
       });
       const data = await result.json();

       dispatch(resetReviewState());

       if (data.clients?.length > 0 && data.clients[0].id !== clientId) {
         navigate(`/review/${data.clients[0].id}`, { replace: true });
       } else {
         navigate('/review', { replace: true });
       }
     } catch {
       // Fallback: just go to queue
       dispatch(resetReviewState());
       navigate('/review', { replace: true });
     }
   };
   ```
